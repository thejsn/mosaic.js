Updating the mosaic

// Init mosaic
var mosaic = new Mosaic(document.getElementById('imagemosaic'));

// Reference to div with images
var imageList = document.getElementById('target-images')

// Add listener
imageList.addEventListener('click', clicked.bind(this), false);

// Set the new mosaic target
function clicked(e) {
  if(!!e.target.src) {
    mosaic.setTargetImage(e.target);
  }
}
